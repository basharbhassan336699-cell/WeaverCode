# 🕸️ weaver-help — دليل أوامر WeaverCode

اقرأ قائمة الأوامر من `.claude/commands/` ثم اعرضها بهذا التنسيق:

---

## 🕸️ WeaverCode — دليل الأوامر الكامل

```
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
           🕸️  W E A V E R C O D E
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📦 الإعداد والتثبيت
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
/weaver-install     التثبيت الأولي الكامل (مرة واحدة فقط)
/weaver-key         تغيير مفتاح API أو المزود
/expand-tools       تثبيت الأدوات الناقصة (44 أداة)

🚀 التشغيل اليومي
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
/weaver-start       إعادة التشغيل بعد إغلاق الجهاز ✨
/weaver-run         تشغيل بوضع أو نموذج محدد
/weaver-status      فحص شامل لحالة النظام

💾 الحفظ والتحديث
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
/weaver-save        حفظ العمل ورفعه لـ GitHub
/weaver-update      تحديث المشروع من GitHub

🧠 الذاكرة والصيانة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
/weaver-memory      إدارة الذاكرة (عرض/بحث/تصدير)
/weaver-fix         إصلاح تلقائي للمشاكل الشائعة
/weaver-help        هذا الدليل 😊

⚙️  التطوير والبناء
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
/build              بناء وتطوير ميزات جديدة
/review             مراجعة الكود
/clone-deps         استنساخ المستودعات المرجعية
/switch-model       تبديل النموذج (داخل Claude Code)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

⚡ سير العمل الموصى به:
┌─────────────────────────────────────────────────┐
│  أول مرة:        /weaver-install               │
│  كل يوم:         /weaver-start                 │
│  بعد العمل:      /weaver-save "وصف"            │
│  مشكلة؟:         /weaver-fix                   │
│  تحديث مفتاح:   /weaver-key                   │
└─────────────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📱 داخل WeaverCode التفاعلي:
  weaver -i             ← تشغيل تفاعلي
  /mode coding          ← وضع البرمجة
  /mode security        ← وضع الأمن
  /model groq           ← تبديل النموذج
  /icon                 ← عرض الأيقونة
  /stats                ← الإحصاءات
  خروج                 ← إنهاء

```
